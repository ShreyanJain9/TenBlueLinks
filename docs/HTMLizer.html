def options_viewer(where, safesearch)
  mab {
    h1 "Your Options"
    p do
      text "Your language and location are currently set to #{Engines.Languages.invert[where]}"
      br
      text "#{if safesearch == "true" then "✅" else "❌" end} SafeSearch" # unicode emojis breaking old browsers
    end
    a href: "/options" do
      text "Edit"
    end
    2.times { br }
    a href: "/" do
      text "Back to Home"
    end
  }
end

def search_form(query, engine_name)
  mab do
    form action: "/results", method: "get" do
      input type: "text", name: "query", size: "40", value: query
      input type: "hidden", name: "engine", value: engine_name
      input type: "submit", value: "Search"
    end
  end
end
