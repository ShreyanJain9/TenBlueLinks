Result = Struct.new(:url, :title, :snippet) do
  # @return [String] An HTML representation of the search result, to be rendered in the results page.
  def to_html
    res = self
    mab do
      a(res.title, href: res.url)
      2.times { br }
      text res.url
      2.times { br }
      text res.snippet
      2.times { br }
    end
  end
end
